# For the variable parch ((number of parents/children aboard the Titanic)) determining if the variable has
# A Strong left Tail OR
# A Strong right Tail OR
# The variable is normally distributed
# Skewness Value will tell the direction as Skewness <0 = Left Skewed/Negative Skew/Left Tail
# Skewness >0 = Right Skewed/Positive Skew/Right Tail
# Skewness = 0 is Normally Distributed (Symmetric)

# Calculating the Skewness
skewness_value = df['Parch'].skew()
print(f"The Skewness of the 'Parch' variable is :{skewness_value:.2f}")
#Interpreting the result
if skewness_value >0:
  print("Interpretation : The distribution is right skewed (Postive Skew/Right Tail)")
elif skewness_value <0:
  print("Intrepretation: The distribution is left skewed (Negative Skew/Left Tail)")
else:
  print("Interpretation: The distribution is normally distributed (Symmetric)")
